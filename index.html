let taskInput = document.getElementById('taskInput'); let addBtn = document.getElementById('addBtn'); let taskList = document.getElementById('taskList'); // Запускаем отрисовку сразу renderTasks(); addBtn.addEventListener('click', function() { let text = taskInput.value.trim(); if (!text) return; let timestamp = Date.now(); // Получаем старые задачи или создаем пустой массив, если их нет let storedData = localStorage.getItem('myTasks'); let tasks = storedData ? JSON.parse(storedData) : []; tasks.push({ text, timestamp }); localStorage.setItem('myTasks', JSON.stringify(tasks)); taskInput.value = ''; renderTasks(); }); function renderTasks() { let storedData = localStorage.getItem('myTasks'); let tasks = storedData ? JSON.parse(storedData) : []; taskList.innerHTML = ''; tasks.forEach(task => { let date = new Date(task.timestamp); let timeStr = date.toLocaleTimeString('ru-RU'); let li = document.createElement('li'); li.textContent = `${task.text} — Сделано в ${timeStr}`; taskList.appendChild(li); }); }
